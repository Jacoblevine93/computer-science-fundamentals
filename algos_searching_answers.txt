Programming Assignment:

1. You and a friend have set a wager to see who can find the word "Albatross" in the dictionary the fastest. Write a program to allow you to win the bet.

A: 

function find(word, array){
    var beginning = 0;
    var end = array.length - 1;
    var middle = Math.floor((beginning + end)/2);
    while(array[middle] !== word && beginning < end){
        if (word < array[middle]){
            end = middle - 1
        } 
        else if (word > array[middle]){
            beginning = middle + 1
        }
        middle = Math.floor((beginning + end)/2);
    }
    return middle
}

var word = 'albatross';
var array = ['a', 'ab', 'ac', 'albatross', 'de', 'df', 'gh', 'lf', 'zc'];

find(word,array);

2. You work at a pet store, and a child has asked you to net the only white-spotted goldfish from the fish tank. Write a program that will help you net the right fish.

A: 

function find(fish,array) {
	for (i=0; i < array.length; i++) {
		if (fish === array[i]) {
			return array[i]
		}
	}
	
}

var fish = 'white spotted goldfish';
var array = ['other', 'other', 'other', 'white spotted goldfish', 'other']

find(fish,array);
